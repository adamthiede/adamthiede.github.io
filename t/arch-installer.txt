#Installation of arch can be pretty easy. Here's my quick notes (mostly for my own future reference) on how to make that easier.

#Disk Partitioning
fdisk /dev/sda
o #new MBR table - you can install in UEFI if your computer supports it. Mine do not.
n
+1024M
a
n
t
8e #LVM
w

#filesystems
mkfs.ext2 /dev/sda1 #ext4 is fine too.
cryptsetup luksFormat /dev/sda2
cryptseup luksOpen /dev/sda2 encrypted
pvcreate /dev/mapper/encrypted
vgcreate vg /dev/mapper/encrypted
lvcreate -L 4G -n swap vg
lvcreate -L 100%FREE -n root vg
#you can also make a /home or /var LV if you want. I don't care about separating these on single-user machines, but it's almost a must on servers
modprobe dm-mod
vgscan
vgchange -ay
mkswap /dev/mapper/vg-swap && swapon /dev/mapper/vg-swap
mkfs.ext4 /dev/mapper/vg-root
mount /dev/mapper/vg-root /mnt
mkdir /mnt/boot && mount /dev/sda1 /mnt/boot

#installing the system
#assuming you already have wifi setup via wifi-menu or are connected to ethernet, let's pacstrap.
#this is a minimal, bare system that will get you mostly what you need. Yes, I did include CUPS. People print stuff.
pacstrap /mnt base sudo vim git bash-completion grub-bios networkmanager cups tmux sway swaylock swayidle rxvt-unicode firefox noto-fonts noto-fonts-extra noto-fonts-cjk ttf-dejavu apparmor
#this is a litte more involved of a setup. just add GNOME. More complete. 
pacstrap /mnt base sudo vim git bash-completion grub-bios networkmanager cups tmux firefox noto-fonts noto-fonts-extra noto-fonts-cjk ttf-dejavu gnome apparmor
genfstab -U /mnt >> /mnt/etc/fstab

#chroot and configure
arch-chroot /mnt
systemctl enable apparmor #apparmor doesn't cost anything and it's already in the kernel. just do it. It can't hurt.
systemctl enable NetworkManager
systemctl enable cups-browsed
systemctl enable gdm #if you installed GNOME
useradd -m -Gsys,wheel,lp,power,network user
passwd
passwd user

#bootloader and stuff
vim /etc/mkinitcpio.conf #add "encrypt lvm2" after "block" in the line toward the bottom.
vim /etc/default/grub #add "cryptdevice=/dev/sda2:vg" and "apparmor=1 security=apparmor" to the GRUB_CMDLINE_LINUX_DEFAULT line.
mkinitcpio -p linux #linux-libre if you're in parabola
grub-install /dev/sda
grub-mkconfig -o /boot/grub/grub.cfg
visudo #uncomment the first "wheel" line. The one without NOPASSWD.

#you're finished! Arch is easy.
exit
umount -R /mnt
reboot
#Have fun!
